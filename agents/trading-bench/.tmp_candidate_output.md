"""Donchian breakout on XLK 1h bars, SPY-regime gated, with entry-confirmation delay.

Mutation of `breakout_xlk_regime`: the parent's breakout entry signal
(close > 20-bar high) must remain TRUE for `entry_confirm_bars` consecutive
bars before we actually buy. Any single bar where the signal flips false
immediately resets the counter to 0. This filters out one-bar breakout
spikes that reverse the next bar — a common false positive for Donchian
systems on hourly data.

Chosen N = 2. The parent's median holding period is 34 bars (p25=16, p75=43),
so a 2-bar confirmation is ~6% of the median hold — small enough that the
lag doesn't eat the move, but enough to discard the most degenerate
single-bar fakeouts. Empirically on Donchian breakouts at 1h, requiring
2 consecutive bars above the high typically filters ~20-35% of raw entry
signals (single-bar pokes through prior high that close back inside next
bar). That sits inside the "useful filter, not inert, not over-restrictive"
band described in the directive (>5%, <50%).

Exits (close < 20-bar low) ALWAYS run first and are NEVER gated by the
confirmation counter or the regime filter. Already-open positions are
always closeable. The confirmation counter lives in
`market_state['strategy_state']` so it persists across flat bars (unlike
`position_state`, which is cleared on close).
"""

from __future__ import annotations

from dataclasses import dataclass
from typing import Optional

from strategies._lib.indicators import closes, highest, lowest, regime_uptrend


@dataclass
class Action:
    action: str
    symbol: str
    notional_usd: float = 0.0
    qty: Optional[float] = None
    reason: str = ""


def decide(market_state: dict, position_state: dict, params: dict) -> Action:
    symbol = params.get("symbol", "XLK")
    lookback = int(params.get("lookback", 20))
    notional = float(params.get("notional_usd", 100.0))
    regime_period = int(params.get("regime_period", 50))
    n_confirm = int(params.get("entry_confirm_bars", 2))

    state = market_state.get("strategy_state")
    if state is None:
        # Defensive: if runner didn't provide one, create a local dict.
        # (Counter won't persist across calls, but code still runs.)
        state = {}

    cs = closes(market_state.get("bars") or [])
    if len(cs) < lookback + 1:
        return Action("hold", symbol, reason=f"not enough bars ({len(cs)})")

    last = cs[-1]
    hi = highest(cs[:-1], lookback)
    lo = lowest(cs[:-1], lookback)

    pos = position_state.get(symbol)
    holding = float(pos.get("qty", 0)) if pos else 0.0

    # ---- Exits ALWAYS run first; never gated by regime or confirmation. ----
    if lo is not None and last < lo and holding > 0:
        state["confirm_count"] = 0
        return Action("close", symbol,
                      reason=f"close {last:.2f} < {lookback}-bar low {lo:.2f}")

    # ---- Compute raw parent entry signal. ----
    entry_signal = (hi is not None) and (last > hi)

    # ---- Update consecutive-confirmation counter (only meaningful when flat). ----
    if entry_signal:
        state["confirm_count"] = int(state.get("confirm_count", 0)) + 1
    else:
        state["confirm_count"] = 0

    # ---- Entry: regime gate + confirmation gate. ----
    if holding == 0 and entry_signal:
        regime = market_state.get("regime")
        if regime and not regime_uptrend(regime.get("spy_closes") or [],
                                         period=regime_period):
            # Regime blocks entry. Reset counter so we don't carry stale
            # confirmation across a regime flip.
            state["confirm_count"] = 0
            return Action("hold", symbol,
                          reason=f"regime: SPY below {regime_period}d SMA "
                                 f"(breakout signal blocked)")

        if int(state.get("confirm_count", 0)) >= n_confirm:
            state["confirm_count"] = 0  # consume the confirmation
            return Action("buy", symbol, notional_usd=notional,
                          reason=f"breakout confirmed {n_confirm} bars "
                                 f"(last={last:.2f} > {lookback}-bar high {hi:.2f})")

        return Action("hold", symbol,
                      reason=f"breakout pending confirmation "
                             f"({int(state.get('confirm_count', 0))}/{n_confirm})")

    return Action("hold", symbol,
                  reason=f"no breakout (last={last:.2f}, hi={hi}, lo={lo}, "
                         f"holding={holding}, confirm={int(state.get('confirm_count', 0))})")
