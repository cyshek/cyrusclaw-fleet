# === strategy.py ===
"""SMA crossover on QQQ 1h bars with a tight intra-trade stop-loss.

Entry: fast SMA crosses above slow SMA while flat (parent signal unchanged).
Exit: parent's crossover-down close ALWAYS runs first. Additionally, we track
the entry price in position_state and force-close if price drops more than
`stop_pct` below entry.

Why 0.80% for the stop? The parent's per-trade drawdown distribution shows
median max-drawdown of ~1.14% and p75 (shallow tail) of ~0.61%. A 0.80% stop
sits between p75 and the median, so it would have triggered on roughly the
worst ~50%+ of trades — catching the intra-trade adverse excursions where
price drifts against us for several bars before the slow SMA finally rolls
over (median holding period is 20 bars; the crossover exit is laggy by
construction). A looser stop (1.5%+) would sit beyond the p25 tail and
essentially never fire — inert code, not edge. A tighter stop (<0.5%) would
fire inside normal 1h noise on nearly every trade and destroy the parent's
trend-capture edge. 0.80% is the sweet spot: tight enough to actually cut
the deep-drawdown tail, loose enough to let typical 1h wiggle breathe.

We persist `entry_price` in position_state[symbol] on the buy bar. On any
close (parent signal or stop), we clear it implicitly by emitting a close
(runner zeroes qty; we re-seed on next buy).
"""

from __future__ import annotations

from dataclasses import dataclass
from typing import Optional

from strategies._lib.indicators import closes, sma


@dataclass
class Action:
    action: str
    symbol: str
    notional_usd: float = 0.0
    qty: Optional[float] = None
    reason: str = ""


def decide(market_state: dict, position_state: dict, params: dict) -> Action:
    symbol = params.get("symbol", "QQQ")
    fast_p = int(params.get("fast", 10))
    slow_p = int(params.get("slow", 30))
    notional = float(params.get("notional_usd", 100.0))
    stop_pct = float(params.get("stop_pct", 0.008))  # 0.80%

    bars = market_state.get("bars") or []
    cs = closes(bars)
    fast = sma(cs, fast_p)
    slow = sma(cs, slow_p)

    pos = position_state.get(symbol) or {}
    holding = float(pos.get("qty", 0)) if pos else 0.0
    entry_price = pos.get("entry_price")
    last = cs[-1] if cs else None

    if fast is None or slow is None:
        return Action("hold", symbol, reason=f"not enough bars ({len(cs)})")

    # Parent exit ALWAYS runs first — stop must not block it.
    if fast < slow and holding > 0:
        return Action("close", symbol,
                      reason=f"SMA{fast_p}={fast:.2f} < SMA{slow_p}={slow:.2f}")

    # Hard stop-loss: close if price dropped > stop_pct below tracked entry.
    if holding > 0 and entry_price is not None and last is not None:
        try:
            ep = float(entry_price)
            if ep > 0 and (last - ep) / ep <= -stop_pct:
                return Action("close", symbol,
                              reason=f"stop-loss {stop_pct*100:.2f}% hit "
                                     f"(last={last:.2f} vs entry={ep:.2f})")
        except (TypeError, ValueError):
            pass

    if fast > slow and holding == 0 and last is not None:
        # Seed entry price into position_state so future bars can stop-out.
        position_state.setdefault(symbol, {})["entry_price"] = float(last)
        return Action("buy", symbol, notional_usd=notional,
                      reason=f"SMA{fast_p}={fast:.2f} > SMA{slow_p}={slow:.2f}")

    return Action("hold", symbol,
                  reason=f"no signal (fast={fast:.2f}, slow={slow:.2f}, holding={holding})")
